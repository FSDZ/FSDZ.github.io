基于Tkinter的聚类分析应用

import tkinter as tk
from tkinter import ttk, filedialog, messagebox # 导入tkinter相关库，用于GUI操作
from PIL import Image, ImageTk # 导入PIL库，用于图片处理
import numpy as np # 导入numpy库，用于数值计算
from scipy.spatial.distance import euclidean # 导入euclidean距离计算函数

class DigitClusteringApp:
def init(self, root):
“””
初始化应用程序。
设置界面标题，大小，并创建界面元素。
“””
self.root = root # Tkinter窗口对象
self.root.title(“聚类分析”) # 窗口标题
self.root.geometry(“1200x800”) # 窗口大小

    # 当前模式：初始化为位图模式
    self.current_mode = "位图"
    self.current_image = None  # 当前显示的图片

    self.create_menu()  # 创建菜单
    self.create_main_interface()  # 创建主界面元素

def create_menu(self):
    """
    创建窗口菜单。
    包括层次聚类菜单，菜单项为最短距离算法。
    """
    menubar = tk.Menu(self.root)  # 创建菜单栏
    self.root.config(menu=menubar)  # 将菜单栏配置到窗口

    # 创建"层次聚类"菜单，并添加到菜单栏
    local_menu = tk.Menu(menubar, tearoff=0)
    menubar.add_cascade(label="层次聚类", menu=local_menu)
    # 添加菜单项"最短距离算法"，并绑定函数
    local_menu.add_command(label="最短距离算法", command=self.run_shortest_distance_clustering)

def create_main_interface(self):
    """
    创建主界面的所有控件，包括模式选择框架、输入面板、输出面板等。
    """
    # 模式选择框架
    mode_frame = ttk.LabelFrame(self.root, text="输入模式")
    mode_frame.pack(fill="x", padx=5, pady=5)

    # 模式标签
    self.mode_label = ttk.Label(mode_frame, text="当前为位图模式")
    self.mode_label.pack(side="left", padx=5)

    # 模式切换按钮
    ttk.Button(mode_frame, text="位图模式", command=self.switch_to_image_mode).pack(side="left", padx=5)
    ttk.Button(mode_frame, text="手写模式", command=self.switch_to_drawing_mode).pack(side="left", padx=5)

    # 主要内容区域
    content_frame = ttk.Frame(self.root)
    content_frame.pack(fill="both", expand=True, padx=5, pady=5)

    # 输入面板
    self.input_frame = ttk.LabelFrame(content_frame, text="输入面板")
    self.input_frame.pack(side="left", fill="both", expand=True, padx=5)

    # 输入画布
    self.input_canvas = tk.Canvas(self.input_frame, bg="white")
    self.input_canvas.pack(fill="both", expand=True)

    # 图片上传按钮
    self.upload_button = ttk.Button(self.input_frame, text="打开一幅图片", command=self.upload_image)
    self.upload_button.pack(pady=5)

    # 输出面板
    output_frame = ttk.LabelFrame(content_frame, text="输出面板")
    output_frame.pack(side="right", fill="both", expand=True, padx=5)

    # 输出画布
    self.output_canvas = tk.Canvas(output_frame, bg="white")
    self.output_canvas.pack(fill="both", expand=True)

    # 清除按钮
    ttk.Button(output_frame, text="清空面板", command=self.clear_output).pack(pady=5)

    # 绘画相关变量
    self.drawing = False
    self.last_x = None
    self.last_y = None

    # 绑定鼠标事件
    self.input_canvas.bind("<Button-1>", self.start_drawing)
    self.input_canvas.bind("<B1-Motion>", self.draw)
    self.input_canvas.bind("<ButtonRelease-1>", self.stop_drawing)

def switch_to_image_mode(self):
    """
    切换到位图模式，显示模式为"位图"，并清空画布。
    """
    self.current_mode = "位图"
    self.mode_label.config(text="当前为位图模式")
    self.clear_input()  # 清空输入画布
    self.upload_button.pack(pady=5)  # 显示上传图片按钮

def switch_to_drawing_mode(self):
    """
    切换到手写模式，显示模式为"手写"，并清空画布。
    """
    self.current_mode = "手写"
    self.mode_label.config(text="当前为手写模式")
    self.clear_input()  # 清空输入画布
    self.upload_button.pack_forget()  # 隐藏上传图片按钮

def upload_image(self):
    """
    打开文件对话框，选择图片文件并显示在输入画布上。
    """
    file_path = filedialog.askopenfilename(
        filetypes=[("Image files", "*.png *.jpg *.jpeg *.gif *.bmp *.ppm *.pgm")]
    )
    if file_path:
        image = Image.open(file_path)  # 打开图片文件
        # 调整图片大小以适应画布
        canvas_width = self.input_canvas.winfo_width()
        canvas_height = self.input_canvas.winfo_height()
        image = image.resize((canvas_width, canvas_height), Image.Resampling.LANCZOS)
        self.current_image = ImageTk.PhotoImage(image)  # 转换为Tkinter可用的图片格式
        self.input_canvas.create_image(0, 0, anchor="nw", image=self.current_image)  # 在画布上显示图片

def start_drawing(self, event):
    """
    开始绘画：记录起始点。
    """
    if self.current_mode == "手写":
        self.drawing = True
        self.last_x = event.x
        self.last_y = event.y

def draw(self, event):
    """
    绘画：在画布上绘制线条，跟踪鼠标的移动。
    """
    if self.current_mode == "手写" and self.drawing:
        self.input_canvas.create_line(
            self.last_x, self.last_y, event.x, event.y,
            width=2, fill="black", smooth=True  # 绘制平滑的线条
        )
        self.last_x = event.x
        self.last_y = event.y

def stop_drawing(self, event):
    """
    停止绘画：结束绘画模式。
    """
    self.drawing = False

def clear_input(self):
    """
    清空输入画布。
    """
    self.input_canvas.delete("all")

def clear_output(self):
    """
    清空输出画布。
    """
    self.output_canvas.delete("all")

def run_shortest_distance_clustering(self):
    """
    运行最短距离算法进行层次聚类。
    如果未上传图片或在手写模式下没有输入内容，则弹出警告。
    """
    if self.current_image is None:
        messagebox.showwarning("警告", "请先上传图片或在手写模式下输入内容！")
        return

    # 模拟样本数据 (实际应用中应从图片提取)
    sample_data = np.random.rand(10, 2)  # 随机生成 10 个点作为示例
    clustered_data = self.shortest_distance_clustering(sample_data)  # 进行层次聚类
    self.display_clustered_data(clustered_data)  # 显示聚类结果

def shortest_distance_clustering(self, data):
    """
    层次聚类的实现：使用最短距离算法合并距离较近的点。
    返回聚类后的数据。
    """
    pattern_num = len(data)  # 数据点数量
    categories = list(range(pattern_num))  # 每个点的初始类别为其索引

    while True:
        min_distance = float('inf')  # 初始化最小距离
        merge_pair = None  # 初始化待合并的点对

        # 计算所有点对之间的欧几里得距离
        for i in range(len(data)):
            for j in range(i + 1, len(data)):
                if categories[i] != categories[j]:
                    distance = euclidean(data[i], data[j])
                    if distance < min_distance:
                        min_distance = distance
                        merge_pair = (categories[i], categories[j])

        # 如果距离小于阈值（0.5），则合并
        if min_distance <= 0.5:
            cat_a, cat_b = sorted(merge_pair)
            for k in range(len(categories)):
                if categories[k] == cat_b:
                    categories[k] = cat_a
        else:
            break  # 距离较远时停止合并

    return list(zip(data, categories))  # 返回点及其所属的类别

def display_clustered_data(self, clustered_data):
    """
    在输出画布上显示聚类结果。
    """
    self.output_canvas.delete("all")  # 清空输出画布
    for point, category in clustered_data:
        x, y = point
        self.output_canvas.create_oval(
            x * 200, y * 200, x * 200 + 10, y * 200 + 10,
            fill=f"#00{category + 2}22", outline=""  # 使用不同的颜色表示不同类别
        )

主函数：启动应用程序。

app = DigitClusteringApp(root) # 创建应用程序实例
